/* Dessin d'un carré et d'un cercle */

var size = 100;
var centerX = 200;
var centerY = 200;
var myCursor : cursor = cursor(centerX, centerY);

myCursor.drawSquare(size);
myCursor.drawCircle(size / 2);

function drawSquare(length) {
    for (var i = 0; i < 4; i = i + 1;) {
        myCursor.drawLine(length, 0);
        myCursor.rotate(90);
    }
}
